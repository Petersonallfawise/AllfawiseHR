function doPost(e) {
  var data = JSON.parse(e.postData.contents);
  var formType = data.formType || '';
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var sheet;

  if (formType === 'live_seller') {
    sheet = ss.getSheetByName('Leads Live Seller');
    sheet.appendRow([
      new Date(),
      data.nome || '',
      data.whatsapp || '',
      data.cidade || '',
      data.estado || '',
      data.idade || '',
      data.comunicativa || '',
      data.tem_celular || '',
      data.trabalhar_casa || '',
      data.experiencia || '',
      data.tiktok || '',
      data.instagram || '',
      data.mensagem || ''
    ]);
  } else if (formType === 'outras_vagas') {
    sheet = ss.getSheetByName('Leads Outras Vagas');
    sheet.appendRow([
      new Date(),
      data.nomeCompleto || '',
      data.whatsapp || '',
      data.email || '',
      data.cidade || '',
      data.estado || '',
      data.vagaEscolhida || '',
      data.disponibilidade || '',
      data.modeloTrabalho || '',
      data.ultimaEmpresa || '',
      data.funcao || '',
      data.atividades || '',
      data.linkedin || '',
      data.nivelIngles || '',
      data.sobreVoce || ''
    ]);
  } else if (formType === 'empresas') {
    sheet = ss.getSheetByName('Leads Empresas');
    sheet.appendRow([
      new Date(),
      data.empresa_nome || '',
      data.responsavel || '',
      data.cargo || '',
      data.email_corporativo || '',
      data.whatsapp_empresa || '',
      data.site_rede || '',
      data.tipo_talento || '',
      data.qtd_vagas || '',
      data.tipo_servico || '',
      data.mensagem_empresa || ''
    ]);
  } else {
    return ContentService
      .createTextOutput(JSON.stringify({ status: 'error', message: 'formType invalido ou ausente' }))
      .setMimeType(ContentService.MimeType.JSON);
  }

  return ContentService
    .createTextOutput(JSON.stringify({ status: 'success' }))
    .setMimeType(ContentService.MimeType.JSON);
}
